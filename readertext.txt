inp1 = input("Input name of your file(XXX.xml):  ")
FILENAME = inp1 

def write():
    message = input("input line: ")
    with open(FILENAME, "a") as file:
        file.write(message + "\n")
def read():
    with open(FILENAME, "r") as file:
        for message in file:
            print(message, end="")
    print()
while(True):
    read()
    selection = int(input("1.input in file\t\t2.read file\t\t3.exit\nselect read: "))
    match selection:
        case 1: write()
        case 2: read()
        case 3: break
        case _: print("error input")
     
print("close program")
