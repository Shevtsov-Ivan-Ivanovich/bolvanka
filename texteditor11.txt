import tkinter as tk
from tkinter import filedialog, messagebox

# Functin for open file
def open_file():
    file_path = filedialog.askopenfilename(filetypes=[("All Files", "*.*")])
    if file_path:
        with open(file_path, "r") as file:
            text.delete(1.0, tk.END)  # clear text field
            text.insert(tk.END, file.read())  # insert text in file

# Functin for saved file
def save_file():
    file_path = filedialog.asksaveasfilename(defaultextension="")
    if file_path:
        with open(file_path, "w") as file:
            file.write(text.get(1.0, tk.END))  # input text in file

# finctin for exit
def exit_app():
    if messagebox.askokcancel("exit", "Are you sure you want to leave?"):
        root.destroy()

# create base window
root = tk.Tk()
root.title("texst editor")

# create base field
text = tk.Text(root, wrap="word")
text.pack(expand=True, fill="both")

# create menu
menu_bar = tk.Menu(root)

# menu "fale"
file_menu = tk.Menu(menu_bar, tearoff=0)
file_menu.add_command(label="open", command=open_file)
file_menu.add_command(label="save", command=save_file)
file_menu.add_separator()
file_menu.add_command(label="exit", command=exit_app)
menu_bar.add_cascade(label="fail", menu=file_menu)

# menu "editing"
edit_menu = tk.Menu(menu_bar, tearoff=0)
edit_menu.add_command(label="copy", command=lambda: text.event_generate("<<Copy>>"))
edit_menu.add_command(label="insert", command=lambda: text.event_generate("<<Paste>>"))
edit_menu.add_command(label="cut", command=lambda: text.event_generate("<<Cut>>"))
menu_bar.add_cascade(label="editing", menu=edit_menu)

#menu "format service"


# add menu in window
root.config(menu=menu_bar)

# start base cikle
root.mainloop()
